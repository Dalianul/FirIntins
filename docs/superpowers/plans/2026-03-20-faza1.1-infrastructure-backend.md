# Faza 1.1 — Infrastructura & Backend Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Set up the Turborepo monorepo root, scaffold the Medusa v2 backend with PostgreSQL/Redis, configure module providers (Stripe, SendGrid), run migrations, create admin user, and seed initial catalog data (region, categories, sample products).

**Architecture:** Turborepo monorepo with pnpm workspaces; `apps/backend` contains the Medusa v2 application. Stripe and SendGrid are registered as module providers nested inside their respective core modules (payment, notification). Seed data is loaded via `medusa exec` using the container directly — no HTTP calls. Docker Compose covers dev services (Varianta A laptop) and production VPS; native PostgreSQL + Memurai covers Windows-native dev (Varianta B PC).

**Tech Stack:** Node.js 20+, pnpm 9+, Turborepo 2, Medusa v2, PostgreSQL 16, Redis 7 / Memurai (Windows), TypeScript strict, Stripe, SendGrid

import { Modules } from "@medusajs/framework/utils"

// This test runs via medusa exec to have container access
// Usage: pnpm medusa exec ./src/scripts/__tests__/seed.test.ts

export default async function seedTest({ container }: ExecArgs) {
  const errors: string[] = []

  // 1. Check Romania region
  const regionService = container.resolve(Modules.REGION)
  const regions = await regionService.listRegions({ name: "Romania" })
  if (regions.length === 0) errors.push("FAIL: Romania region not found")
  else if (regions[0].currency_code !== "ron")
    errors.push("FAIL: Romania region currency is not RON")
  else console.log("PASS: Romania region exists with RON currency")

  // 2. Check product categories
  const productService = container.resolve(Modules.PRODUCT)
  const categories = await productService.listProductCategories()
  const categoryNames = categories.map((c: any) => c.name)
  const expected = ["Lansete", "Mulinete", "Fire", "Carlige", "Accesorii", "Boilies & Momeli"]
  for (const name of expected) {
    if (!categoryNames.includes(name))
      errors.push(`FAIL: Category "${name}" not found`)
    else console.log(`PASS: Category "${name}" exists`)
  }

  // 3. Check sample products
  const [products] = await productService.listAndCountProducts({
    title: "Lanseta Crap Pro 3.6m",
  })
  if (products.length === 0) errors.push("FAIL: Lanseta Crap Pro 3.6m not found")
  else {
    console.log("PASS: Lanseta Crap Pro 3.6m exists")
    const opts = await productService.listProductOptions({
      product_id: products[0].id,
    })
    if (!opts.some((o: any) => o.title === "TC"))
      errors.push("FAIL: Lanseta missing TC option")
    else console.log("PASS: Lanseta has TC option")
  }

  if (errors.length > 0) {
    console.error("\nTest failures:")
    errors.forEach((e) => console.error(e))
    process.exit(1)
  } else {
    console.log("\nAll seed tests passed!")
  }
}

import { Modules, ContainerRegistrationKeys } from "@medusajs/framework/utils"

  const regionService = container.resolve(Modules.REGION)

  const productService = container.resolve(Modules.PRODUCT)

  // 1. Region Romania
  logger.info("Creating Romania region...")
  const [region] = await regionService.createRegions([
    {
      name: "Romania",
      currency_code: "ron",
      countries: ["ro"],
    },
  ])
  logger.info(`Region created: ${region.id}`)

  // 2. Sales Channel
  logger.info("Creating sales channel...")
  const [salesChannel] = await salesChannelService.createSalesChannels([
    {
      name: "Magazin Online",
