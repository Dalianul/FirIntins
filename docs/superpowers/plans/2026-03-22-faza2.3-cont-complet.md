# Cont Complet (Faza 2.3) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Complete the customer account section with four fully functional pages: Profile (view/edit name + email), Addresses (list, add, edit, delete, set default), Security (change password via two-step re-auth), and Orders (list with detail link).

**Architecture:** All pages are Server Components under `app/cont/` protected by `middleware.ts` (existing `_medusa_jwt` cookie check). Data fetched server-side via Medusa SDK. Mutations via Server Actions (Zod-validated, call `medusa.store.customer.*`). Password change: re-auth via `medusa.auth.login()` to get fresh token → update with fresh token. Default address stored in `customer.metadata.default_address_id`. Account nav gets 4 new links. No new backend modules needed — uses Medusa's built-in Customer module.

**Tech Stack:** Next.js 16 App Router, Medusa SDK (store customer endpoints), TypeScript, Zod, Jest + ts-jest

// apps/storefront/actions/account.ts
"use server"

import { cookies } from "next/headers"
import { revalidatePath } from "next/cache"
import { medusa } from "@/lib/medusa/client"
import {
  UpdateProfileSchema,
  AddressSchema,
  ChangePasswordSchema,
  UpdateProfileInput,
  AddressInput,
  ChangePasswordInput,
} from "@/lib/schema/account"

async function getAuthOptions(): Promise<{ headers: Record<string, string> } | null> {
  const cookieStore = await cookies()
  const token = cookieStore.get("_medusa_jwt")?.value
  if (!token) return null
  return { headers: { Authorization: `Bearer ${token}` } }
}

export async function updateProfileAction(input: UpdateProfileInput) {
  const parsed = UpdateProfileSchema.safeParse(input)
  if (!parsed.success) {
    return { success: false, error: parsed.error.errors[0]?.message ?? "Date invalide" }
  }

  const auth = await getAuthOptions()
  if (!auth) return { success: false, error: "Trebuie să fii autentificat" }

  try {
    await medusa.store.customer.update(parsed.data, auth)
    revalidatePath("/cont/profil")
    return { success: true }
  } catch {
    return { success: false, error: "Nu am putut actualiza profilul." }
  }
}

export async function addAddressAction(input: AddressInput) {
  const parsed = AddressSchema.safeParse(input)
  if (!parsed.success) {
    return { success: false, error: parsed.error.errors[0]?.message ?? "Date invalide" }
  }

  const auth = await getAuthOptions()
  if (!auth) return { success: false, error: "Trebuie să fii autentificat" }

  try {
    const data = await medusa.store.customer.createAddress(parsed.data, auth)
    revalidatePath("/cont/adrese")
    return { success: true, address: data.address }
  } catch {
    return { success: false, error: "Nu am putut adăuga adresa." }
  }
}

export async function updateAddressAction(addressId: string, input: AddressInput) {
  const parsed = AddressSchema.safeParse(input)
  if (!parsed.success) {
    return { success: false, error: parsed.error.errors[0]?.message ?? "Date invalide" }
  }

  const auth = await getAuthOptions()
  if (!auth) return { success: false, error: "Trebuie să fii autentificat" }

  try {
    await medusa.store.customer.updateAddress(addressId, parsed.data, auth)
    revalidatePath("/cont/adrese")
    return { success: true }
  } catch {
    return { success: false, error: "Nu am putut actualiza adresa." }
  }
}

export async function deleteAddressAction(addressId: string) {
  const auth = await getAuthOptions()
  if (!auth) return { success: false, error: "Trebuie să fii autentificat" }

  try {
    await medusa.store.customer.deleteAddress(addressId, auth)
    revalidatePath("/cont/adrese")
    return { success: true }
  } catch {
    return { success: false, error: "Nu am putut șterge adresa." }
  }
}

export async function setDefaultAddressAction(addressId: string) {
  const auth = await getAuthOptions()
  if (!auth) return { success: false, error: "Trebuie să fii autentificat" }

  try {
    await medusa.store.customer.update(
      { metadata: { default_address_id: addressId } },
      auth
    )
    revalidatePath("/cont/adrese")
    return { success: true }
  } catch {
    return { success: false, error: "Nu am putut seta adresa implicită." }
  }
}

export async function changePasswordAction(input: ChangePasswordInput) {
  const parsed = ChangePasswordSchema.safeParse(input)
  if (!parsed.success) {
    return { success: false, error: parsed.error.errors[0]?.message ?? "Date invalide" }
  }

  const auth = await getAuthOptions()
  if (!auth) return { success: false, error: "Trebuie să fii autentificat" }

  try {
    // Step 1: Get current customer email
    const customerData = await medusa.client.fetch<{ customer: { email: string } }>(
      "/store/customers/me",
      auth
    )
    const email = customerData.customer.email

    // Step 2: Re-authenticate with current password to get fresh token
    const freshToken = await medusa.auth.login("customer", "emailpass", {
      email,
      password: parsed.data.current_password,
    })

    if (!freshToken || typeof freshToken !== "string") {
      return { success: false, error: "Parola curentă este incorectă" }
    }

    // Step 3: Update password using fresh token
    await medusa.store.customer.update(
      { password: parsed.data.new_password },
      { headers: { Authorization: `Bearer ${freshToken}` } }
    )

    return { success: true }
  } catch (error) {
    if (error instanceof Error && error.message.toLowerCase().includes("invalid")) {
      return { success: false, error: "Parola curentă este incorectă" }
    }
    return { success: false, error: "Nu am putut schimba parola. Încearcă din nou." }
  }
}

// apps/storefront/app/cont/profil/profile-form.tsx
"use client"

import { useState } from "react"
import { updateProfileAction } from "@/actions/account"

interface ProfileFormProps {
  customer: {
    first_name?: string
    last_name?: string
    email: string
    phone?: string
  }
}

export default function ProfileForm({ customer }: ProfileFormProps) {
  const [firstName, setFirstName] = useState(customer.first_name ?? "")
  const [lastName, setLastName] = useState(customer.last_name ?? "")
  const [phone, setPhone] = useState(customer.phone ?? "")
  const [pending, setPending] = useState(false)
  const [success, setSuccess] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    if (pending) return
    setPending(true)
    setSuccess(false)
    setError(null)

    const result = await updateProfileAction({
      first_name: firstName,
      last_name: lastName,
      phone: phone || undefined,
    })

    setPending(false)
    if (result.success) {
      setSuccess(true)
    } else {
      setError(result.error ?? "Eroare")
    }
  }

  const fieldClass =
    "w-full bg-color-surface border border-white/10 text-color-cream px-3 py-2 font-outfit text-sm focus:outline-none focus:border-color-moss/50"
  const labelClass = "block text-color-fog font-outfit text-xs uppercase tracking-wider mb-1"

  return (
    <form onSubmit={handleSubmit} className="space-y-5">
      <div className="grid grid-cols-2 gap-4">
        <div>
          <label className={labelClass}>Prenume</label>
          <input
            type="text"
            value={firstName}
            onChange={(e) => setFirstName(e.target.value)}
            required
            className={fieldClass}
          />
        </div>
        <div>
          <label className={labelClass}>Nume</label>
          <input
            type="text"
            value={lastName}
            onChange={(e) => setLastName(e.target.value)}
            required
            className={fieldClass}
          />
        </div>
      </div>

      <div>
        <label className={labelClass}>Email</label>
        <input
          type="email"
          value={customer.email}
          disabled
          className={`${fieldClass} opacity-50 cursor-not-allowed`}
        />
        <p className="text-color-fog/50 font-outfit text-xs mt-1">
          Emailul nu poate fi modificat din această pagină.
        </p>
      </div>

      <div>
        <label className={labelClass}>Telefon (opțional)</label>
        <input
          type="tel"
          value={phone}
          onChange={(e) => setPhone(e.target.value)}
          className={fieldClass}
        />
      </div>

      {success && (
        <p className="text-color-moss font-outfit text-sm">Profilul a fost actualizat cu succes.</p>
      )}
      {error && <p className="text-red-400 font-outfit text-sm">{error}</p>}

      <button
        type="submit"
        disabled={pending}
        className="px-6 py-3 bg-color-moss text-white font-outfit text-sm hover:bg-color-moss-light transition-colors disabled:opacity-50"
      >
        {pending ? "Se salvează..." : "Salvează modificările"}
      </button>
    </form>
  )
}

// apps/storefront/app/cont/securitate/change-password-form.tsx
"use client"

import { useState } from "react"
import { changePasswordAction } from "@/actions/account"

export default function ChangePasswordForm() {
  const [currentPassword, setCurrentPassword] = useState("")
  const [newPassword, setNewPassword] = useState("")
  const [confirmPassword, setConfirmPassword] = useState("")
  const [pending, setPending] = useState(false)
  const [success, setSuccess] = useState(false)
  const [error, setError] = useState<string | null>(null)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    if (pending) return
    setPending(true)
    setSuccess(false)
    setError(null)

    const result = await changePasswordAction({
      current_password: currentPassword,
      new_password: newPassword,
      confirm_password: confirmPassword,
    })

    setPending(false)

    if (result.success) {
      setSuccess(true)
      setCurrentPassword("")
      setNewPassword("")
      setConfirmPassword("")
    } else {
      setError(result.error ?? "Eroare")
    }
  }

  const inputClass =
    "w-full bg-color-surface border border-white/10 text-color-cream px-3 py-2 font-outfit text-sm focus:outline-none focus:border-color-moss/50"
  const labelClass = "block text-color-fog font-outfit text-xs uppercase tracking-wider mb-1"

  return (
    <form onSubmit={handleSubmit} className="space-y-5">
      <div>
        <label className={labelClass}>Parola curentă</label>
        <input
          type="password"
          value={currentPassword}
          onChange={(e) => setCurrentPassword(e.target.value)}
          required
          className={inputClass}
          autoComplete="current-password"
        />
      </div>

      <div>
        <label className={labelClass}>Parola nouă</label>
        <input
          type="password"
          value={newPassword}
          onChange={(e) => setNewPassword(e.target.value)}
          required
          minLength={8}
          className={inputClass}
          autoComplete="new-password"
        />
        <p className="text-color-fog/50 font-outfit text-xs mt-1">Minim 8 caractere.</p>
      </div>

      <div>
        <label className={labelClass}>Confirmă parola nouă</label>
        <input
          type="password"
          value={confirmPassword}
          onChange={(e) => setConfirmPassword(e.target.value)}
          required
          className={inputClass}
          autoComplete="new-password"
        />
      </div>

      {success && (
        <p className="text-color-moss font-outfit text-sm">
          Parola a fost schimbată cu succes.
        </p>
      )}
      {error && <p className="text-red-400 font-outfit text-sm">{error}</p>}

      <button
        type="submit"
        disabled={pending}
        className="px-6 py-3 bg-color-moss text-white font-outfit text-sm hover:bg-color-moss-light transition-colors disabled:opacity-50"
      >
        {pending ? "Se procesează..." : "Schimbă parola"}
      </button>
    </form>
  )
}
