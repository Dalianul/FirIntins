# Faza 3.6 — Analytics & GA4 Design

**Date:** 2026-04-01
**Status:** Approved
**Scope:** GA4 ecommerce event tracking with Consent Mode v2, wired to existing cookie consent

---

## Overview

Integrate Google Analytics 4 with Consent Mode v2 into the Next.js 16 storefront. GA4 loads immediately with denied defaults, then upgrades to granted when the user accepts analytics via the existing `vanilla-cookieconsent` banner. Four core ecommerce events are tracked: `view_item`, `add_to_cart`, `begin_checkout`, `purchase`.

The Measurement ID is supplied via `NEXT_PUBLIC_GA_MEASUREMENT_ID`. When the variable is absent, analytics is a no-op — no errors, no network requests.

---

## Architecture

### Approach: Custom `<Analytics>` component + `lib/analytics.ts` utility

No new dependencies. Uses `next/script` for script injection and plain `gtag` calls for events. The existing `vanilla-cookieconsent` `onConsent`/`onChange` callbacks handle consent state transitions.

## Section 1: Script Injection + Consent Mode v2

**`components/analytics/analytics.tsx`** — Server Component:

```tsx
import Script from "next/script"

const GA_ID = process.env.NEXT_PUBLIC_GA_MEASUREMENT_ID

export function Analytics() {
  if (!GA_ID) return null

  return (
    <>
      <Script
        src={`https://www.googletagmanager.com/gtag/js?id=${GA_ID}`}
        strategy="afterInteractive"
      />
      <Script id="ga4-init" strategy="afterInteractive">
        {`
          window.dataLayer = window.dataLayer || [];
          function gtag(){dataLayer.push(arguments);}
          window.gtag = gtag;
          gtag('consent', 'default', {
            analytics_storage: 'denied',
            ad_storage: 'denied',
            wait_for_update: 500
          });
          gtag('js', new Date());
          gtag('config', '${GA_ID}');
        `}
      </Script>
    </>
  )
}
```

The inline init script runs before `gtag.js` processes any hits. `wait_for_update: 500` gives `vanilla-cookieconsent` 500ms to restore a prior consent choice on page load — preventing a denied ping for returning users who previously accepted.

Added to `app/(main)/layout.tsx` immediately before `<CookieConsent />`.

## Section 2: `lib/analytics.ts` — Typed Event Helpers

```ts
import type { Cart, CartItem } from "@/context/cart-context"

interface GA4Item {
  item_id: string
  item_name: string
  item_variant?: string
  price: number
  quantity: number
}

function gtag(...args: unknown[]): void {
  if (typeof window === "undefined" || !window.gtag) return
  window.gtag(...args)
}

// Consent management
export function grantAnalyticsConsent(): void {
  gtag("consent", "update", {
    analytics_storage: "granted",
    ad_storage: "granted",
  })
}

export function denyAnalyticsConsent(): void {
  gtag("consent", "update", {
    analytics_storage: "denied",
    ad_storage: "denied",
  })
}

// GA4 ecommerce events
export function trackViewItem(product: {
  id: string
  title: string
  price?: number
}): void {
  gtag("event", "view_item", {
    currency: "RON",
    value: product.price ? product.price / 100 : 0,
    items: [
      {
        item_id: product.id,
        item_name: product.title,
        price: product.price ? product.price / 100 : 0,
        quantity: 1,
      },
    ],
  })
}

export function trackAddToCart(item: CartItem): void {
  const price = item.unit_price / 100
  gtag("event", "add_to_cart", {
    currency: "RON",
    value: price * item.quantity,
    items: [
      {
        item_id: item.variant_id,
        item_name: item.product_title,
        item_variant: item.variant_title ?? undefined,
        price,
        quantity: item.quantity,
      },
    ],
  })
}

export function trackBeginCheckout(cart: Cart): void {
  const items: GA4Item[] = cart.items.map((item) => ({
    item_id: item.variant_id,
    item_name: item.product_title,
    item_variant: item.variant_title ?? undefined,
    price: item.unit_price / 100,
    quantity: item.quantity,
  }))
  gtag("event", "begin_checkout", {
    currency: "RON",
    value: cart.total / 100,
    items,
  })
}

export function trackPurchase(
  orderId: string,
  total: number,
  items: GA4Item[]
): void {
  gtag("event", "purchase", {
    transaction_id: orderId,
    currency: "RON",
    value: total / 100,
    items,
  })
}
```

All functions guard with `typeof window === "undefined" || !window.gtag` — safe in SSR and when GA4 is not loaded (missing env var).

Prices stored as cents in Medusa, converted to RON (÷100) before sending to GA4.

## Section 3: Cookie Consent Integration

Modify `components/cookie-consent/cookie-consent.tsx` — add two callbacks to the existing `CookieConsent.run()` config:

```ts
import { grantAnalyticsConsent, denyAnalyticsConsent } from "@/lib/analytics"

// Inside CookieConsent.run({ ... }):

onConsent: ({ cookie }) => {
  if (cookie.categories.includes("analytics")) {
    grantAnalyticsConsent()
  }
},

onChange: ({ cookie }) => {
  if (cookie.categories.includes("analytics")) {
    grantAnalyticsConsent()
  } else {
    denyAnalyticsConsent()
  }
},
```

- **`onConsent`** — fires on every page load when a prior choice is restored from localStorage. Re-grants analytics for returning users who previously accepted without re-showing the banner.
- **`onChange`** — fires when user changes preference in the settings modal. Handles both grant and revoke.

No other changes to `cookie-consent.tsx`.

## Section 4: Event Wiring

### `view_item` — Product Detail Page

Create `components/analytics/view-item-tracker.tsx`:

```tsx
"use client"

import { useEffect } from "react"
import { trackViewItem } from "@/lib/analytics"

interface ViewItemTrackerProps {
  id: string
  title: string
  price?: number
}

export function ViewItemTracker({ id, title, price }: ViewItemTrackerProps) {
  useEffect(() => {
    trackViewItem({ id, title, price })
  }, [id, title, price])

  return null
}
```

In `app/(main)/(shop)/produse/[handle]/page.tsx` (Server Component), add after the product data fetch:

```tsx
<ViewItemTracker
  id={product.id}
  title={product.title}
  price={product.variants?.[0]?.prices?.[0]?.amount}
/>
```

### `add_to_cart` — Cart Context

In `context/cart-context.tsx`, inside `addItem()`, after the Medusa call succeeds:

```ts
import { trackAddToCart } from "@/lib/analytics"

// After: const updatedCart = await addItemToCart(...)
// The updated cart contains the newly added item. Find it by variant_id:
const addedItem = updatedCart?.items?.find((i) => i.variant_id === variantId)
if (addedItem) trackAddToCart(addedItem)
```

### `begin_checkout` — Checkout Client

In `components/checkout/checkout-client.tsx`, add a second `useEffect` that fires once when cart is available:

```ts
import { useRef } from "react"
import { trackBeginCheckout } from "@/lib/analytics"

const checkoutTracked = useRef(false)

useEffect(() => {
  if (cart && !checkoutTracked.current) {
    checkoutTracked.current = true
    trackBeginCheckout(cart)
  }
}, [cart])
```

`useRef` prevents double-fire in React Strict Mode.

### `purchase` — Order Confirmation

Create `components/analytics/purchase-tracker.tsx`:

```tsx
"use client"

import { useEffect, useRef } from "react"
import { trackPurchase } from "@/lib/analytics"

interface PurchaseTrackerItem {
  item_id: string
  item_name: string
  item_variant?: string
  price: number
  quantity: number
}

interface PurchaseTrackerProps {
  orderId: string
  total: number
  items: PurchaseTrackerItem[]
}

export function PurchaseTracker({ orderId, total, items }: PurchaseTrackerProps) {
  const tracked = useRef(false)

  useEffect(() => {
    if (!tracked.current) {
      tracked.current = true
      trackPurchase(orderId, total, items)
    }
  }, [orderId, total, items])

  return null
}
```

In `app/(main)/(checkout)/checkout/confirmare/[orderId]/page.tsx`, map the Medusa order to tracker props and render `<PurchaseTracker>`. The order object from Medusa contains `id`, `total`, and `items` array with `variant_id`, `product_title`, `variant_title`, `unit_price`, `quantity`.
